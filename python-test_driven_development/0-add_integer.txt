#!/usr/bin/python3
import unittest
add_integer = __import__('0-add_integer').add_integer

class TestAddInteger(unittest.TestCase):
    def test_integers(self):
        self.assertEqual(add_integer(2, 3), 5)
        self.assertEqual(add_integer(-5, 5), 0)
        self.assertEqual(add_integer(100, -50), 50)

    def test_default_b(self):
        self.assertEqual(add_integer(2), 100)   # 2 + 98 = 100

    def test_floats(self):
        self.assertEqual(add_integer(2.7, 3.5), 5)   # 2 + 3
        self.assertEqual(add_integer(-1.9, 2.9), 1) # -1 + 2

    def test_type_errors(self):
        with self.assertRaises(TypeError):
            add_integer("3", 5)
        with self.assertRaises(TypeError):
            add_integer(2, "5")
        with self.assertRaises(TypeError):
            add_integer(None, 5)
        with self.assertRaises(TypeError):
            add_integer([1, 2], 3)
        with self.assertRaises(TypeError):
            add_integer(2, { "a": 1 })

    def test_large_numbers(self):
        self.assertEqual(add_integer(10**6, 10**6), 2000000)

if __name__ == "__main__":
    unittest.main()
